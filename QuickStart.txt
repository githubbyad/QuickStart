    <script>
        var t = document.querySelectorAll(`#quickstart textarea`);
        if (t.length > 0) {
            for (var u = 0; u < t.length; u++) {
                t[u].outerHTML = `<section data-block="${t[u].getAttribute('data-block')}" data-location="${t[u].getAttribute('data-location')}"><textarea>${t[u].innerHTML}</textarea></section>`;
            }
        }
        var q = document.querySelectorAll(`#quickstart section`);
        if (q.length > 0) {
            for (var i = 0; i < q.length; i++) {
                q[i].insertAdjacentHTML(`afterbegin`, `<area title="${q[i].getAttribute('data-block')}" shape="react" coords="${q[i].getAttribute('data-location')}">`)
                q[i].addEventListener(`mouseover`, function() {
                    var coords = this.getAttribute(`data-location`).split(`,`);
                    var width = document.getElementsByClassName(`quickstart-container`)[0].offsetWidth;
                    var height = document.getElementsByClassName(`quickstart-container`)[0].offsetHeight;
                    document.getElementsByClassName(`map-selector`)[0].classList.add(`hover`);
                    document.getElementsByClassName(`map-selector`)[0].setAttribute(`style`, `left:${coords[0]}px;top:${coords[1]}px;right:${width - coords[2]}px;bottom:${height - coords[3]}px;`);
                }, function() {
                    document.getElementsByClassName(`map-selector`)[0].classList.remove(`hover`);
                    document.getElementsByClassName(`map-selector`)[0].setAttribute(`style`, ``);
                });
            }
        }
        var ba = document.querySelectorAll(`area`);
        if (ba.length > 0) {
            for (x = 0; x < ba.length; x++) {
                ba[x].addEventListener(`click`, function() {
                    document.getElementById(`block-modal`).innerHTML = `<button type='button' class='block-button btn btn-primary d-none' data-bs-toggle='modal' data-bs-target='#staticBackdrop'>Steps</button> <div class='modal fade' id='staticBackdrop' data-bs-backdrop='static' data-bs-keyboard='false' tabindex='-1' aria-labelledby='staticBackdropLabel' aria-hidden='true'> <div class='modal-dialog modal-dialog-scrollable modal-dialog-centered'> <div class='modal-content'> <div class='modal-header bg-primary'> <h5 class='modal-title fw-bold text-light w-100 text-center' id='staticBackdropLabel'>${this.title}</h5> <button type='button' class='btn-close' data-bs-dismiss='modal' aria-label='Close'></button> </div> <div class='modal-body'>${this.nextElementSibling.value.trim()}</div> </div> </div> </div>`;
                    document.getElementsByClassName(`block-button`)[0].click();
                });
                ba[x].addEventListener(`mouseover`, function() {
                    var y = this.title;
                    setTimeout(function() {
                        document.getElementById(`block-css`).innerHTML = ``;
                        document.getElementById(`block-css`).insertAdjacentHTML(`beforeend`, `<style>.map-selector.hover::after{content:"${y}"}</style>`);
                    }, 200);
                });
            }
        }
    </script>
